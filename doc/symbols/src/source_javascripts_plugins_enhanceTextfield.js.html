1 (function($, ME){
  2   var isNeutralKey = ME.util.isNeutralKey,
  3   isRemovalKey = ME.util.isRemovalKey;
  4 
  5   $.fn.enhanceTextfield = function(options){
  6     options = options || {};
  7 
  8     return this.each(function(){
  9       var $t = $(this), $btn, keyPressed,
 10       defaultColor = $t.css("color");
 11 
 12       function displayPrompt(){
 13         $t.css("color","grey").data('hasPrompt', true).val(options.prompt);
 14       }
 15 
 16       function clearField(){
 17         $t.css("color", defaultColor).data('hasPrompt', false).val('');
 18       }
 19 
 20       function checkState(){
 21         if($t.val() && $t.val() !== options.prompt){
 22           $t.css("color", defaultColor);
 23           $btn.show();
 24         } else {
 25           displayPrompt();
 26           $btn.hide();
 27           keyPressed = false;
 28         }
 29       }
 30 
 31       if($t.is("input")){
 32         $btn = $("<span>x</span>").click(function(){
 33           clearField();
 34           $t.focus();
 35           $btn.hide();
 36           keyPressed = false;
 37         });
 38         $p = $t.wrap("<span class=\"clearButton\">")
 39           .focus(function(){
 40             if(!keyPressed && $t.val() === options.prompt){
 41               clearField();
 42             }
 43             $t.parent().addClass('focus');
 44           })
 45           .focusout(function(){
 46             if(!keyPressed && !$t.val()){
 47               displayPrompt();
 48             }
 49             $t.parent().removeClass('focus');
 50           })
 51           .keydown(function(e){
 52             if(!isNeutralKey(e.which) &&
 53                // User may try to delete the prompt. Help him with that
 54                (!keyPressed || isRemovalKey(e.which)) &&
 55                $t.val() === options.prompt){
 56               clearField();
 57             }
 58           })
 59           .keyup(function(e){
 60             if(!keyPressed && (
 61               !isNeutralKey(e.which) &&
 62                 // Prevent reapearing of the prompt after it has just
 63                 // been deleted
 64                 !isRemovalKey(e.which)
 65             )){
 66               keyPressed = true;
 67               checkState();
 68             }
 69           })
 70           .bind("blur change",function(){
 71             checkState();
 72           })
 73           .submit(function(){
 74             if($t.val() === options.prompt){
 75               clearField();
 76             }
 77           })
 78           .parent().append($btn);
 79         if($t.hasClass("ui-corner-left")){
 80           $p.addClass("ui-corner-left");
 81         } else {
 82           $p.addClass("ui-corner-all");
 83         }
 84       }
 85     });
 86   };
 87 })(jQuery, ME);
