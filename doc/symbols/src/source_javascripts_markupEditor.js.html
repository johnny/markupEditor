179     /**
180      * @param {String} [boundary] The right and left boundary the
181      * selection should be extended to
182      * @returns {String} The currently selected string
183      */
184     getSelection: function(boundary) {
185       var textArea = this.textArea, text = textArea.val(), boundaryPosition, subString;
186       textArea.focus();
187 
188       // gecko & webkit
189       this.scrollPosition = textArea.scrollTop;
190       this.selectionStart = textArea[0].selectionStart;
191       this.selectionEnd = textArea[0].selectionEnd;
192 
193       if(text[this.selectionEnd-1] === "\n"){
194         this.selectionEnd -= 1;
195       }
196 
197       if(boundary) {
198         // find left boundary
199         boundaryPosition = Math.max(text.lastIndexOf(boundary, this.selectionStart), text.lastIndexOf("\n", this.selectionStart));
200         if(boundaryPosition !== -1) {
201           this.selectionStart = boundaryPosition + 1;
202         } else {
203           this.selectionStart = 0;
204         }
205         
206         // find right boundary, first limit the text to the
207         // next new line
208         boundaryPosition = text.indexOf("\n", this.selectionEnd); 
209         if(boundaryPosition === -1) {
210           subString = text.slice(this.selectionStart);
211         } else {
212           subString = text.slice(this.selectionStart, boundaryPosition);
213         }
214 
215         // Then find the next boundary
216         boundaryPosition = 0;
217         do{
218           boundaryPosition = subString.indexOf(boundary, boundaryPosition + 1);
219         } while(boundaryPosition !== -1 && this.selectionEnd > this.selectionStart + boundaryPosition);
220 
221         // when it doesn't exist, extend the selection to the
222         // paragraph end
223         if(boundaryPosition === -1) {
224           boundaryPosition = subString.length;
225         }
226         this.selectionEnd = this.selectionStart + boundaryPosition;
227       }
228       this.selection = text.slice(this.selectionStart, this.selectionEnd);
229       return this.selection;
230     },
231     /**

237       var textArea = this.textArea,
238       newSelectionStart = this.selectionStart,
239       newSelectionEnd = this.selectionStart + string.length;

241       // gecko & webkit
242       textArea.val(textArea.val().slice(0, this.selectionStart) + string + textArea.val().slice(this.selectionEnd, textArea.val().length));

247       } else if(collapseToStart === false){

249       }
250 
251       textArea[0].setSelectionRange(newSelectionStart, newSelectionEnd);
252       textArea.focus();
253     },

255      * Extend the right selection with a regexp. Everything matched will be added
256      * to the selection. Useful for special cases like toggling parts of a bolded
257      * String in textile
258      * 
259      * @param {Regexp} regexp The regexp
260      * 
261      * @example
262      * extendRightSelection(/ +/)
263      */

428     /**
429      * Load the toolbar for the current mode. If a toolbar item is not
430      * supported, it will be hidden.
431      */
432     loadModeToolbar: function(){
433       var supportedItems = this.editor.currentMode.supportedItems,
434       hasSave = this.editor.settings.save,
435       oldVisibleItems = this.visibleItems,
436       newVisibleItems = [];
437       
438       // Optimize: better scheme. Calculate the differences between

440       this.div.children().each(function(){
441         var item = this.className;

443           if(!oldVisibleItems || oldVisibleItems.indexOf(item) == -1){
444             $(this).show();
445           }
446           newVisibleItems.push(item);
447         } else {
448           if(!oldVisibleItems || oldVisibleItems.indexOf(item) != -1){
449             $(this).hide();
450           }
451         }
452       });
453       this.visibleItems = newVisibleItems;
454     },

457      *

459      * @param {HTMLElement} target The target of the click

462       var item = toolbarItems[action],
463       editor = this.editor,
464       mode = editor.currentMode;
465       (item[mode.id] || item).clicked(editor, mode, target);
466       // Update Preview in case something has changed

468         mode.updatePreview();
469       }
470     },

472      * Activate the buttons/selects of the given actions on the toolbar
473      * 
474      * @param {Object} actions The actions which should be active

478       if(actions) {
479         this.div.children().each(function(i) {
480           var action = this.className.split(" ")[0];
481           if (actions[action] == 'disable') { // deactivate
482             this.disabled = true;
483             this.className = action + " disabled";
484           } else {
485             this.disabled = false;
486             this.className = action;
487             if(actions[action] === true) { // buttons
488               this.className = action + " on";
489             } else if(actions[action]){ // selects
490               this.value = actions[action];
491             }
492           }
493         });
494       }

598      * @param {String} modeId The id of the mode (e.g. textile)

601      */

603       if(this.loadedModes[modeId]) {
604         return this.loadedModes[modeId];
605       }
606       else if (availableModes[modeId]) {
607         this.loadedModes[modeId] = availableModes[modeId](this);
608         return this.loadedModes[modeId];
609       }
610       else {
611         console.log("Mode " + modeId + " is not defined");

616       cssClasses = classString.split(/\s+/);

620

623         }
624       }
