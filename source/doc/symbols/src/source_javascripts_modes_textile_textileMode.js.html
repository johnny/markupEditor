  6    * CONSIDER rename or move setParagraphs out of it
  7    *
  8    * @param {Editor} editor The editor to work on
  9    * @param {Function} functor The functor will be applied on each paragraph
 10    */
 11   function eachParagraph(editor, functor) {
 12     var i,
 13     paragraphs = textileMode.getParagraphs(editor),
 14     paragraphsLength = paragraphs.length;
 15 
 16     for(i = 0; i < paragraphsLength; i++) {
 17       paragraphs[i] = functor(paragraphs[i]);
 18     }
 19

 21   }

 23   /**
 24    * Apply the functor to each given line
 25    *
 26    * @param {String[]} lines The line array
 27    * @param {Function} functor The function which will be applied to
 28    * each line
 29    */
 30   function eachLine(lines, functor){
 31     var linesLength = lines.length,
 32     i, line, lineStart, match;
 33     for(i = 0; i < linesLength; i++) {
 34       line = lines[i];
 35 
 36       // Skip blank lines
 37       if( !/^\s*$/.test(line)) {
 38         match = line.match(/^((?:\w+\. )?(?: *[\*#] )?)\s*(.*)/);
 39         lineStart = match[1];
 40         line = match[2];
 41 
 42         functor(i, lineStart, line);
 43       }
 44     }
 45   }

 47   /**
 48    * Apply the functor to each line of the current selection.
 49    *
 50    * @param {Editor} editor The editor to work on
 51    * @param {String} boundary The boundary to which the Selection will
 52    * be extended e.g. '\n', ' '
 53    * @param {Function} functor The function to apply to each line
 54    */
 55   function replaceEachLine(editor, boundary, functor){
 56     var lines = textileMode.extendSelection(editor, boundary).split("\n");
 57 
 58     eachLine(lines, function(i, lineStart, line){
 59       lines[i] = functor(lineStart, line);
 60     });
 61     
 62     textileMode.replaceSelection(editor, lines.join("\n"));
 63   }

 65   /**
 66    * Get the first line of the selection, without the special textile
 67    * tokens which might be at the start of the line.
 68    *
 69    * @param {Editor} editor The editor to work on
 70    *
 71    * @returns {String} The first line of the selection
 72    */
 73   function firstLine(editor, boundary){
 74     // lines has only one element
 75     var lines = textileMode.extendSelection(editor, boundary).split("\n").slice(0,1),
 76     lineLength = lines[0].length;
 77     editor.selectionEnd = editor.selectionStart + lineLength;
 78     // Ignore special textile tokens at the beginnig of the line
 79     eachLine(lines, function(i, lineStart, line){
 80       editor.setSelectionRange( editor.selectionStart + lineStart.length, editor.selectionEnd);
 81       lines[i] = line;
 82     });
 83     return lines[0];
 84   }

 86   /**
 87    * Execute align command
 88    * 
 89    * @param {Editor} editor The editor to work on
 90    * @param {String} orientation The orientation of the alignment
 91    */
 92   function align(editor, orientation) {
 93     eachParagraph(editor, function(paragraph) {
 94       var classes, classesLength, newClasses = [];
 95 
 96       if(/^\w+\([^)]+\)\./.test(paragraph)) {
 97         classes = jQuery.trim(paragraph.slice(paragraph.indexOf("(") + 1, paragraph.indexOf(")"))).split(/\s+/);
 98         classesLength = classes.length;
 99         for(i=0 ; i < classesLength ; i++ ){
100           if(classes[i] != 'right' && classes[i] != 'left' && classes[i] != 'center'){
101             newClasses.push(classes[i]);
102           }
103         }
104         newClasses.push(orientation);
105         return paragraph.replace(/^(\w+)[^.]+.\s+/, "$1(" + newClasses.join(" ") + "). ");
106       } else if(/^\w+\./.test(paragraph)) {
107         return paragraph.replace(/^(\w+)\.\s*/, "$1(" + orientation + "). ");
108       } else {
109         return "p(" + orientation + "). " + paragraph;
110       }
111     });
112   }

114   /**
115    * Scan the textarea for the first match and set selection to it.
116    * This is useful e.g. for finding a link markup with a given source
117    * 
118    * @param {Editor} editor The editor to work on
119    * @param {RegExp} r The regexp to search for
120    */
121   function scanForMatch(editor, r){
122     var text = editor.textArea.val(),
123     match = r.exec(text);
124     if(r.lastIndex === 0){
125       return;   // TODO escalate this return to break the caller too
126     }
127     while(r.lastIndex < editor.selectionStart){
128       match = r.exec(text);
129     }
130 
131     // needed for the replaceSelection call
132     editor.setSelectionRange(r.lastIndex - match[0].length, r.lastIndex);
133 
134     return match;
135   }

179       }

181       return false;

382     /**
383      * Convert preview div to textile
384      * 
385      * @param {Editor} editor The editor to work on
386      *
387      * @returns {String} A textile string
388      */
389     toText: function(editor, callback) {
390       var html = editor.preview.html();
391 
392       /**
393        * Fetch the regexps for the given tags and call the given
394        * callback
395        *
396        * TODO get rid of the tags parameter?
397        */
398       function eachRegexp(tags, callback){
399         var i, item,
400         items = {
401           b: [/(\s*)<(?:b|strong)>((?:.|[\r\n])*?)<\/(?:b|strong)>(\s*)/gi,'*'],
402           i: [ /(\s*)<(?:i|em)>((?:.|[\r\n])*?)<\/(?:i|em)>(\s*)/gi, '_'],
403           del: [ /(\s*)<(?:strike|del)>((?:.|[\r\n])*?)<\/(?:strike|del)>(\s*)/gi, '-'],
404           u: [ /(\s*)<(?:u|ins)>((?:.|[\r\n])*?)<\/(?:u|ins)>(\s*)/gi, '+']
405         };
406         for(i = tags.length; i; i--){
407           item = items[tags[i-1]];
408           callback(item[0], item[1]);
409         }
410       }
411 
412       html = html.replace(/\s*<(ul|ol)>((.|[\r\n])*?)<\/\1>\s*/gi, function(match, tag, items){
413         var bullet = tag == 'ul' ? '*' : '#';
414         
415         eachRegexp(['b','i', 'u', 'del'], function(regexp, delimiter){
416           items = items.replace(regexp, function(match, startSpace, text, endSpace){
417             startSpace = startSpace ? ' ' : '';
418             endSpace = endSpace ? ' ' : '';
419             return startSpace + delimiter + text + delimiter + endSpace;
420           });
421         });
422 
423         return items.replace(/\s*<li>((.|[\r\n])*?)<\/li>\s*/gi, bullet + " $1\n") + "\n";
424       });
425       
426       html = html.replace(/ *<(p|h[1-4])([^>]*)>((.|[\r\n])*?)<\/\1>\s*/gi, function(match, tag, attributes, content){
427         var front = "", cssClass = attributes.match(/class=\"([^"]*)/);
428         if(cssClass){
429           front = tag + "(" + cssClass[1] + "). ";
430         } else if(tag != "p"){
431           front = tag + ". ";
432         }
433         eachRegexp(['b','i', 'u', 'del'], function(regexp, d){
434           content = content.replace(regexp, function(match, startSpace, text, endSpace){
435             startSpace = startSpace ? ' ' : '';
436             endSpace = endSpace ? ' ' : '';
437             return startSpace + d + text.replace(/<br ?\/?>\s*/gi, d + "\n" + d) + d + endSpace;
438           });
439         });
440         
441         return front + content.replace(/<br ?\/?>\s*/gi, "\n") + "\n\n";
442       });
443       
444       html = html.replace(/\s*<img[^>]*>\s*/gi, function(match){
445         var img = $(match),
446         replacement = img.attr('src'),
447         title = img.attr('title');
448 
449         if(title && !/^\s*$/.test(title)){
450           replacement = replacement + "(" + title + ")";
451         }
452         return "!" + replacement + "!";
453       });
454       html = html.replace(/(\s*)<a href="([^\"]*)">((?:.|[\r\n])*?)<\/a>(\s*)/gi, function(match, startSpace, uri, content, endSpace){
455         var out = startSpace ? ' ' : '';
456 
457         if(/^\s*![^!]+!\s*$/.test(content)){
458           out += $.trim(content) + ":";
459         } else {
460           out += "\"" + content + "\":";
461         }
462         out += uri;
463         if(endSpace){
464           out += ' ';
465         }
466         return out;
467       });
468       html = html.replace(/\s*<code[^>]*>((.|[\r\n])*?)<\/code>\s*/gi, ' @$1@ ');
469       html = html.replace(/(\r\n|\n){3,}/g, "\n\n");
470       html = html.replace(/ /g, ' ');
471       html = html.replace(/^[\r\n]+|[\r\n]+$/g, '');
472 
473       return html;
474     },

526       startOfParagraphs = editor.startOfParagraphs;
