  4   /**
  5    * Iterate over each paragraph and call the functor on it and set the paragraphs
  6    * CONSIDER rename or move setParagraphs out of it
  7    *
  8    * @param {Editor} editor The editor to work on
  9    * @param {Function} functor The functor will be applied on each paragraph
 10    */
 11   function eachParagraph(editor, functor) {
 12     var paragraphs = textileMode.getParagraphs(editor), paragraphsLength = paragraphs.length;
 13 
 14     for(i = 0; i < paragraphsLength; i++) {
 15       paragraphs[i] = functor(paragraphs[i]);
 16     }
 17     textileMode.setParagraphs(editor, paragraphs);
 18   }

 20   /**
 21    * Apply the functor to each given line
 22    *
 23    * @param {String[]} lines The line array
 24    * @param {Function} functor The function which will be applied to
 25    * each line
 26    */
 27   function eachLine(lines, functor){
 28     var linesLength = lines.length,
 29     i, line, lineStart, match;
 30     for(i = 0; i < linesLength; i++) {
 31       line = lines[i];
 32 
 33       // Skip blank lines
 34       if( !/^\s*$/.test(line)) {
 35         match = line.match(/^((?:\w+\. )?(?: *[\*#] )?)\s*(.*)/);
 36         lineStart = match[1];
 37         line = match[2];
 38 
 39         functor(i, lineStart, line);
 40       }
 41     }
 42   }

 44   /**
 45    * Apply the functor to each line of the current selection.
 46    *
 47    * @param {Editor} editor The editor to work on
 48    * @param {String} boundary The boundary to which the Selection will
 49    * be extended e.g. '\n', ' '
 50    * @param {Function} functor The function to apply to each line
 51    */
 52   function replaceEachLine(editor, boundary, functor){
 53     var lines = textileMode.extendSelection(editor, boundary).split("\n");
 54 
 55     eachLine(lines, function(i, lineStart, line){
 56       lines[i] = functor(lineStart, line);
 57     });
 58     
 59     textileMode.replaceSelection(editor, lines.join("\n"));
 60   }

 62   /**
 63    * Get the first line of the selection, without the special textile
 64    * tokens which might be at the start of the line.
 65    *
 66    * @param {Editor} editor The editor to work on
 67    *
 68    * @returns {String} The first line of the selection
 69    */
 70   function firstLine(editor, boundary){
 71     // lines has only one element
 72     var lines = textileMode.extendSelection(editor, boundary).split("\n").slice(0,1),
 73     lineLength = lines[0].length;
 74     editor.selectionEnd = editor.selectionStart + lineLength;
 75     // Ignore special textile tokens at the beginnig of the line
 76     eachLine(lines, function(i, lineStart, line){
 77       editor.setSelectionRange( editor.selectionStart + lineStart.length, editor.selectionEnd);
 78       lines[i] = line;
 79     });
 80     return lines[0];
 81   }

 83   /**
 84    * Execute align command
 85    * 
 86    * @param {Editor} editor The editor to work on
 87    * @param {String} orientation The orientation of the alignment
 88    */
 89   function align(editor, orientation) {
 90     eachParagraph(editor, function(paragraph) {
 91       var classes, classesLength, newClasses = [];
 92       if(/^\w+\([^)]+\)\./.test(paragraph)) {
 93         classes = jQuery.trim(paragraph.slice(paragraph.indexOf("(") + 1, paragraph.indexOf(")"))).split(/\s+/);
 94         classesLength = classes.length;
 95         for(i=0 ; i < classesLength ; i++ ){
 96           if(classes[i] != 'right' && classes[i] != 'left' && classes[i] != 'center'){
 97             newClasses.push(classes[i]);
 98           }
 99         }
100         newClasses.push(orientation);
101         return paragraph.replace(/^(\w+)[^.]+.\s+/, "$1(" + newClasses.join(" ") + "). ");
102       } else if(/^\w+\./.test(paragraph)) {
103         return paragraph.replace(/^(\w+)\.\s*/, "$1(" + orientation + "). ");
104       } else {
105         return "p(" + orientation + "). " + paragraph;
106       }
107     });
108   }

110   /**
111    * Scan the textarea for the first match and set selection to it.
112    * This is useful e.g. for finding a link markup with a given source
113    * 
114    * @param {Editor} editor The editor to work on
115    * @param {RegExp} r The regexp to search for
116    */
117   function scanForMatch(editor, r){
118     var text = editor.textArea.val(),
119     match = r.exec(text);
120     if(r.lastIndex === 0){
121       return;   // TODO escalate this return to break the caller too
122     }
123     while(r.lastIndex < editor.selectionStart){
124       match = r.exec(text);
125     }
126 
127     // needed for the replaceSelection call
128     editor.setSelectionRange(r.lastIndex - match[0].length, r.lastIndex);
129 
130     return match;
131   }

164   function pressedEnter(editor){
165     var list = editor.currentNodes.list, replacement;
166 
167     if(list && /(u|o)l/i.test(list.tag)){ // only headings
168       textileMode.getSelection(editor);
169       if(ME.holdShift){
170         replacement = " <br> ";
171       } else {
172         replacement = "\n" + listTypes[list.tag] + " ";
173       }
174       textileMode.replaceSelection(editor, replacement, false);
175       return false;
176     }

387       /**
388        * Fetch the regexps for the given tags and call the given
389        * callback
390        *
391        * TODO get rid of the tags parameter?
392        */
393       function eachRegexp(tags, callback){
394         var i, item,
395         items = {
396           b: [/<(?:b|strong)>((.|[\r\n])*?)<\/(?:b|strong)>/gi,'*'],
397           i: [ /<(?:i|em)>((.|[\r\n])*?)<\/(?:i|em)>/gi, '_'],
398           del: [ /<(?:strike|del)>((.|[\r\n])*?)<\/(?:strike|del)>/gi, '-'],
399           u: [ /<(?:u|ins)>((.|[\r\n])*?)<\/(?:u|ins)>/gi, '+']
400         };
401         for(i = tags.length; i; i--){
402           item = items[tags[i-1]];
403           callback(item[0], item[1]);
404         }
405       }
406 
407       html = html.replace(/\s*<(ul|ol)>((.|[\r\n])*?)<\/\1>\s*/gi, function(match, tag, items){
408         var bullet = tag == 'ul' ? '*' : '#';
409         
410         eachRegexp(['b','i', 'u', 'del'], function(regexp, delimiter){
411           items = items.replace(regexp, delimiter + '$1' + delimiter);
412         });
413 
414         return items.replace(/\s*<li>((.|[\r\n])*?)<\/li>\s*/gi, bullet + " $1\n") + "\n";
415       });
416       
417       html = html.replace(/ *<(p|h[1-4])([^>]*)>((.|[\r\n])*?)<\/\1>\s*/gi, function(match, tag, attributes, content){
418         var front = "", cssClass = attributes.match(/class=\"([^"]*)/);
419         if(cssClass){
420           front = tag + "(" + cssClass[1] + "). ";
421         } else if(tag != "p"){
422           front = tag + ". ";
423         }
424 
425         eachRegexp(['b','i', 'u', 'del'], function(regexp, d){
426           content = content.replace(regexp, function(match, text){
427             return d + text.replace(/<br ?\/?>\s*/gi, d + "\n" + d) + d;
428           });
429         });
430         
431         return front + content.replace(/<br ?\/?>\s*/gi, "\n") + "\n\n";
432       });
433       
434       html = html.replace(/<img[^>]*>/gi, function(match){
435         var img = $(match),
436         replacement = img.attr('src'),
437         title = img.attr('title');
438 
439         if(title && !/^\s*$/.test(title)){
440           replacement = replacement + "(" + title + ")";
441         }
442         return "!" + replacement + "!";
443       });
444       html = html.replace(/<a href="([^\"]*)">((.|[\r\n])*?)<\/a>/gi, function(match, uri, content){
445         if(/^\s*![^!]+!\s*$/.test(content)){
446           return $.trim(content) + ":" + uri;
447         } else {
448           return "\"" + content + "\":" + uri;
449         }
450       });
451       html = html.replace(/\s*<code[^>]*>((.|[\r\n])*?)<\/code>\s*/gi, ' @$1@ ');
452       html = html.replace(/(\r\n|\n){3,}/g, "\n\n");
453       html = html.replace(/ /g, ' ');
454       html = html.replace(/^[\r\n]+|[\r\n]+$/g, '');
455 
456       return html;
457     },

481     /**
482      * Set the paragraphs and move the caret
483      * 
484      * @param {Editor} editor The editor to work on
485      * @param {String[]} paragraphs An array of paragraphs
486      */
487     setParagraphs: function(editor, paragraphs) {
488       var text = editor.textArea.val();
489       paragraphs = paragraphs.join("\n\n");
490 
491       if(editor.rightBoundary === -1) {
492         editor.textArea.val(text.slice(0,editor.leftBoundary) + paragraphs);
493       } else {
494         editor.textArea.val(text.slice(0,editor.leftBoundary) + paragraphs + text.slice(editor.rightBoundary));
495       }
496       
497       this.moveCaret(editor, paragraphs.length - editor.boundaryDistance);
498     },
499     /**
500      * Move the caret by the given distance. Positive values move the caret to 
501      * the right, negative to the left.
502      * 
503      * @param {Editor} editor The editor to work on
504      * @param {Integer} distance The distance to move the caret
505      */
506     moveCaret: function(editor, distance) {
507       // console.log("Moving caret: " + distance);
508       var selectionStart = editor.selectionStart,
509       startOfParagraphs = editor.startOfParagraphs;
510 
511       if(Math.abs(selectionStart - startOfParagraphs) > Math.abs(distance)) {
512         selectionStart += distance;
513       } else {
514         selectionStart = startOfParagraphs;
515       }
516       
517       editor.textArea.focus();
518       editor.setSelectionRange(selectionStart, selectionStart);
519     },
