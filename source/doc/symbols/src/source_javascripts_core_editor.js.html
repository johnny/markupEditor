 20   ME.Editor = function Editor(textArea, settings) {

 22 
 23     activeEditors[numberOfEditors] = editor;
 24     editor.id = numberOfEditors;
 25     // console.log("init editor " + numberOfEditors);

 27

 33     function addKeyListeners(object, isTextarea){
 34       object.keydown(function(e){
 35         if(isTextarea || editor.is('wysiwyg')){
 36           return editor.currentMode.pressed(editor, e.keyCode);
 37         }
 38       }).keyup(function(e){
 39         if(isTextarea || editor.is('wysiwyg')){
 40           return editor.currentMode.released(editor, e.keyCode);
 41         }
 42       }).mouseup(function(){
 43         if(isTextarea || editor.is('wysiwyg')){
 44           editor.focus();
 45           return editor.currentMode.clicked(editor);
 46         }
 47       });
 48     }
 49 
 50     // mouseup will catch all three mouse buttons. Since all keys move
 51     // the cursor a check is necessary
 52     this.textArea = textArea.bind("mouseup keyup", function() {
 53       editor.checkState();
 54       clearTimeout(timer);
 55       timer = setTimeout(function(){
 56         editor.currentMode.updatePreview(editor);
 57       },1000);
 58     });
 59     addKeyListeners(textArea,true);
 60 
 61     if(preview){
 62       preview.addClass('preview');
 63     } else {
 64       preview = $("<div class=\"preview\">");
 65     }
 66     this.preview = preview.bind("mouseup keyup", function() {
 67       if(editor.is("wysiwyg")) {
 68         editor.checkState();
 69       }
 70     });
 71     addKeyListeners(this.preview);
 72 
 73     this.overlay = $('<div class=\"overlay\"><div class=\"background\"></div><div class=\"spinner\"></div></div>');
 74     
 75     this.toolbar = new ME.Toolbar(this);
 76     this.container = textArea.wrap("<div class=\"markupEditor\">")

 79       .append(this.overlay)
 80       .prepend(this.toolbar.div);

 84   /**
 85    * Extract the datatype from the given class string. It looks for
 86    * known modes or valid options in the given select
 87    *
 88    * @param {String} classString The css class string to scan
 89    * @param {HTMLSelect} [select] The select with all available modes
 90    * (even undefined ones, which will be created on the fly)
 91    *
 92    * @returns {String|undefined} The mode id of the data mode, if it exists
 93    */
 94   ME.Editor.extractDataType = function(classString, select){
 95     var i, cssClass,
 96     cssClasses = classString.split(/\s+/);
 97 
 98     for(i = 0; i < cssClasses.length; i += 1) {
 99       cssClass = cssClasses[i];
100 
101       if(cssClass !== "wysiwyg" && ME.hasMode(cssClass)) {
102         return cssClass;
103       }
104     }

116       oldMode = editor.currentMode;

122

133           });

137     /**
138      * Change the current underlying data format
139      * 
140      * @param {String} modeId The id of the mode (e.g. textile)
141      */
142     changeDataMode: function(modeId, silent){
143       var isInWysiwyg = this.is('wysiwyg'),
144       dataMode = ME.getMode(modeId);
145       if(!modeId || modeId === this.currentMode.id) {
146         return false;
147       }
148 
149       if(isInWysiwyg && dataMode.toText){
150         this.dataType = modeId;
151         this.checkState();
152         this.syncEditors(silent);
153       } else {
154         this.changeMode(modeId);
155       }
156     },
157     warnIfNecessary: function(oldMode, nextMode, callback, silent){
158       var editor = this;
159       if(silent || (nextMode.toText && oldMode.toHTML)){
160         callback();
161       } else {
162         // TODO use i18n shortcut here
163         if(nextMode.toText){
164           text = 'noticeMissingToHTML';
165         } else {
166           text = 'noticeMissingToText';
167         }
168         dialogProxy = ME.dialog.notice(['Ok', 'Cancel'], _(text));
169         dialogProxy.dialog('open', {
170           submit: function(){
171             if(callback){
172               callback();
173             }
174           },
175           cancel: function(){
176             editor.toolbar.div.find('.changeDataMode')
177               .val(oldMode.id);
178           }
179         });
180       }
181     },
182     /**
183      * Lock the Editor, idempotent
184      */
185     beginModeChange: function(){
186       this.overlay.show();
187     },
188     syncEditors: function(silent) {
189       var i, editorID, otherEditor, editorIDs,
190       externalSelect = this.settings.select,
191       newModeID = this.currentMode.id;
192 
193       if(externalSelect && !silent){
194         editorIDs = externalSelect.data('editorIDs') || [];
195 
196         for(i = editorIDs.length; --i >= 0;){
197           editorID = editorIDs[i];
198           if(editorID === this.id){
199             continue;
200           }
201 
202           otherEditor = activeEditors[editorID];
203 
204           if(newModeID !== otherEditor.currentMode.id){
205             otherEditor.changeMode(newModeID, true);
206           } else { // wysiwyg datamode change
207             otherEditor.changeDataMode(this.dataType, true);
208           }
209         }
210       }
