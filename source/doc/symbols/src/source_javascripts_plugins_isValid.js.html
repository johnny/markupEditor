 20   function Validator($form){
 21     this.$form = $form;
 22     
 23     //Get all the required inputs
 24     this.inputs = $form.find(':input.required');
 25   }
 26 
 27   Validator.prototype = {
 28     check: function() {
 29       var valid = true;
 30 
 31       this.reset(true);
 32       
 33       this.inputs.each(function() {
 34         var $input = $(this),
 35         fieldValue = $.trim($input.val()),
 36         labelText = $input.parent().prev().text(),
 37         errorMsg;
 38         
 39         //Check if it's empty or an invalid email
 40         if(fieldValue === '') {
 41           errorMsg = _('errorText') || opts.errorText;
 42 	  valid = false;
 43         } else if($input.hasClass('email')) {
 44 	  if(!(/^([_a-z0-9-]+)(\.[_a-z0-9-]+)*@([a-z0-9-]+)(\.[a-z0-9-]+)*(\.[a-z]{2,4})$/.test(fieldValue))) {
 45 	    errorMsg = _('emailErrorText') || opts.emailErrorText;
 46 	    valid = false;
 47 	  }
 48         }
 49         
 50         //If there is an error, display it
 51         if(errorMsg) {
 52           errorMsg = errorMsg.replace('{label}',labelText);
 53           $input.parent()
 54             .addClass(opts.errorClass)
 55             .error('show', errorMsg);
 56         }
 57       });
 58 
 59       return valid;
 60     },
 61     reset: function(fadeOut){
 62       return this.inputs.each(function(){
 63         $(this).parent()
 64           .removeClass(opts.errorClass)
 65           .error('hide', fadeOut);
 66       });
 67     }
 68   };

 70   function get($element, constructor, dataField){
 71     var data = $element.data(dataField);
 72     if (!data) {
 73       data = new constructor($element);
 74       $element.data(dataField, data);
 75     }
 76     return data;
 77   }
 78   /**
 79    * Validates forms
 80    *
 81    * @memberOf jQuery.isValid.prototype
 82    *
 83    * @param {String} [method="check"] The method to call
 84    */
 85   $.fn.isValid = function(method, options) {
 86     var valid = true;
 87 
 88     this.each(function(){
 89       var validator = get( $(this), Validator, 'isValid');
 90       
 91       if (!validator[method]) {
 92         method = 'check';
 93       }
 94       
 95       valid = validator[method](options) ? valid : false;
 96     });
 97     
 98     return valid;
 99   };

116   function Error($element) {
117     this.$element = $element;
118     this.$tip = $('<div class="error-tip"></div>')
119       .html('<div class="arrow"></div><div class="inner"></div>')
120       .prependTo(document.body);
121   }

123   Error.prototype = {
124     opacity: 0.8,
125     show: function(message) {
126       var $tip = this.$tip;
127       this.hidden = false;
128       
129       $tip.find('.inner').text(message);
130       $tip.css({top: 0, left: 0, visibility: 'hidden', display: 'block'});
131       
132       var pos = $.extend({}, this.$element.offset(), {
133         width: this.$element[0].offsetWidth,
134         height: this.$element[0].offsetHeight
135       });
136       
137       var actualHeight = $tip[0].offsetHeight;
138       
139       $tip.css({
140         top: pos.top + pos.height / 2 - actualHeight / 2,
141         left: pos.left + pos.width
142       });
143 
144       $tip.stop().css({opacity: 0, display: 'block', visibility: 'visible'})
145         .animate({opacity: this.opacity});
146     },
147     
148     hide: function(fadeOut) {
149       var that = this;
150       if(this.hidden){
151         return;
152       }
153       if (fadeOut) {
154         this.$tip.stop().fadeOut(function() { that.hidden = true; });
155       } else {
156         this.$tip.hide();
157         this.hidden = true;
158       }
159     }
