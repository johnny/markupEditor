1 (function($) {
  2   // TODO cache fields
  3   var opts, errorMsgType, initialized,
  4   methods = {
  5     check: function(options) {
  6       var $this = this, valid = true;
  7       //Hide any errors that are already showing
  8       $this.find(opts.errorElement + '.' + opts.errorClass).remove();
  9       $this.find(':input.' + opts.inputErrorClass).removeClass(opts.inputErrorClass);
 10 
 11       //Get all the required inputs
 12       $this.find(':input.required').each(function() {
 13         var $input = $(this),
 14         fieldValue = $.trim($input.val()),
 15         labelText = $input.siblings('label').text().replace(opts.removeLabelChar, ''),
 16         errorMsg = '';
 17         
 18         //Check if it's empty or an invalid email
 19         if(fieldValue === '') {
 20 	  errorMsg = hasLabelPlaceholder ? errorMsg = opts.errorText.replace('{label}',labelText) : errorMsg = opts.errorText;
 21 	  valid = false;
 22         } else if($input.hasClass('email')) {
 23 	  if(!(/^([_a-z0-9-]+)(\.[_a-z0-9-]+)*@([a-z0-9-]+)(\.[a-z0-9-]+)*(\.[a-z]{2,4})$/.test(fieldValue))) {
 24 	    errorMsg = hasLabelPlaceholder ? errorMsg = opts.emailErrorText.replace('{label}',labelText) : errorMsg = opts.emailErrorText;
 25 	    valid = false;
 26 	  }
 27         }
 28         
 29         //If there is an error, display it
 30         if(errorMsg !== '') {
 31           $input.parent().addClass(opts.errorClass);
 32 	  //$input.addClass(opts.inputErrorClass).after('<'+opts.errorElement+' class="'+opts.errorClass+'">' + errorMsg + '</'+opts.errorElement+'>');
 33         }
 34       });
 35 
 36       return valid;
 37     },
 38     reset: function(){
 39       return this.find(':input.required').each(function(){
 40         $(this).parent().removeClass(opts.errorClass);
 41       });
 42     }
 43   };
 44 
 45   /**
 46    * Validates forms
 47    *
 48    * @memberOf jQuery.prototype
 49    *
 50    * @param {String} [method="init"] The method to call
 51    */
 52   $.fn.isValid = function(method) {
 53     if (!methods[method]) {
 54       method = 'check';
 55     }
 56     return methods[method].apply(this, Array.prototype.slice.call(arguments, 1));
 57   };
 58   
 59   $.fn.isValid.init = function(options){
 60     opts = $.extend({}, $.fn.isValid.defaults, options);
 61     hasLabelPlaceholder = opts.errorText.indexOf("{label}") > -1;
 62   };
 63 
 64   // default options
 65   $.fn.isValid.defaults = {
 66     errorClass: 'error',
 67     errorText: '{label} is a required field.',
 68     emailErrorText: 'Please enter a valid {label}',
 69     errorElement: 'strong',
 70     removeLabelChar: '*'
 71   };
 72 })(jQuery);
