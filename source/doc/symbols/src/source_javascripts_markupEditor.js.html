178     /**
179      * @param {String} [boundary] The right and left boundary the
180      * selection should be extended to
181      * @returns {String} The currently selected string
182      */
183     getSelection: function(editor, boundary) {
184       var textArea = editor.textArea, text = textArea.val(), boundaryPosition, subString;
185       textArea.focus();
186 
187       // gecko & webkit
188       editor.initSelectionProperties();
189 
190       // TODO does this introduce some edgecases? Eating newlines
191       // move this to initSelectionProperties?
192       if(text[editor.selectionEnd-1] === "\n"){
193         editor.selectionEnd -= 1;
194       }
195 
196       if(boundary) {
197         // find left boundary
198         boundaryPosition = Math.max(text.lastIndexOf(boundary, editor.selectionStart), text.lastIndexOf("\n", editor.selectionStart));
199         if(boundaryPosition !== -1) {
200           editor.boundaryStart = boundaryPosition + 1;
201         } else {
202           editor.boundaryStart = 0;
203         }
204         
205         // find right boundary, first limit the text to the
206         // next new line
207         boundaryPosition = text.indexOf("\n", editor.selectionEnd); 
208         if(boundaryPosition === -1) {
209           subString = text.slice(editor.boundaryStart);
210         } else {
211           subString = text.slice(editor.boundaryStart, boundaryPosition);
212         }
213 
214         // Then find the next boundary
215         boundaryPosition = 0;
216         do{
217           boundaryPosition = subString.indexOf(boundary, boundaryPosition + 1);
218         } while(boundaryPosition !== -1 && editor.selectionEnd > editor.boundaryStart + boundaryPosition);
219 
220         // when it doesn't exist, extend the selection to the
221         // paragraph end
222         if(boundaryPosition === -1) {
223           boundaryPosition = subString.length;
224         }
225         editor.boundaryEnd = editor.boundaryStart + boundaryPosition;
226       }
227       editor.boundaryDistance = boundaryPosition;
228       return text.slice(editor.boundaryStart, editor.boundaryEnd);
229     },

241       var textArea = editor.textArea,

245       // gecko & webkit

251       } else if(collapseToStart === false){

253       }
254 
255       editor.setSelectionRange(newSelectionStart, newSelectionEnd);
256       textArea.focus();
257     },

260      * to the selection. Useful for special cases like toggling parts of a bolded
261      * String in textile
262      * 
263      * @param {RegExp} regexp The regexp
264      * 
265      * @example
266      * extendRightSelection(/ +/)

297       }

391    * Create a toolbar for an editor. Every editor has its own toolbar, since the
392    * items of the toolbar can be defined on a per editor basis (save callback)
393    *
394    * @constructor
395    * @name Toolbar
396    */
397   function Toolbar(editor) {
398 
399     // init Toolbar Items
400     var button, buttonTags = '',
401     toolbarDiv = $("<div class=\"toolbar\"></div>"),
402     that = this;
403 
404     this.div = toolbarDiv;
405 
406     toolbarDiv.html(getToolbarHTML());
407 
408     toolbarDiv.mouseup(function(e) { // Trigger on button click
409       var target = e.target;
410 
411       if((/(a|span)/i).test(target.nodeName)) {
412         // When the span is clicked change the Target to the
413         // containing div
414         if(/span/i.test(target.nodeName)) {
415           target = target.parentNode;
416         }
417         if(target.disabled){
418           // TODO handle focus somewhere
419           if(editor.is('wysiwyg')){
420             editor.htmlDiv.focus();
421           } else {
422             editor.textArea.focus();
423           }
424           return false;
425         }
426         var action = target.className;
427 
428         action = action.split(" ")[0];
429         that.runAction(editor, action, target);
430         // TODO this does not work with dialogs
431         // in dialogs this gets set manually, but perhaps there is a
432         // more general way?
433         editor.checkState();
434       }
435     }).change(function(e) { // trigger on select change
436       var target = e.target;
437       that.runAction(editor, target.className, target);
438       return false;
439     }).click(function(e){return false; }); //

442   Toolbar.prototype = /** @scope Toolbar.prototype */{
443     /**
444      * Load the toolbar for the current mode. If a toolbar item is not
445      * supported, it will be hidden.
446      */
447     loadModeToolbar: function(editor){
448       var supportedItems = editor.currentMode.supportedItems,
449       oldVisibleItems = this.visibleItems,
450       newVisibleItems = [];
451       
452       // Optimize: better scheme. Calculate the differences between
453       // the modes once and use them here
454       this.div.children().each(function(){
455         var item = this.className.split(' ')[0], isAvailable = toolbarItems[item].isAvailable;
456         if(supportedItems.indexOf(item) != -1 && (!isAvailable || isAvailable(editor))){
457           if(!oldVisibleItems || oldVisibleItems.indexOf(item) == -1){
458             $(this).show();
459           }
460           newVisibleItems.push(item);
461         } else {
462           if(!oldVisibleItems || oldVisibleItems.indexOf(item) != -1){
463             $(this).hide();
464           }
465         }
466       });
467       this.visibleItems = newVisibleItems;
468     },
469     /**
470      * Execute the given action of the current mode
471      * 
472      * @param {String} action The action to execute
473      * @param {HTMLElement} target The target of the click
474      */
475     runAction: function(editor,action,target) {
476       var item = toolbarItems[action],
477       mode = editor.currentMode;
478 
479       // execute buttons clicked action
480       (item[mode.id] || item).clicked(editor, target);
481       
482       // Update Preview in case something has changed
483       if(action != "changeMode" && !editor.is("wysiwyg")) {
484         mode.updatePreview(editor);
485       }
486     },
487     /**
488      * Activate the buttons/selects of the given actions on the toolbar
489      * 
490      * @param {Object} actions The actions which should be active
491      */
492     setActive: function( actions ) {
493       // activate each action in actions
494       if(actions) {
495         this.div.children().each(function(i) {
496           var action = this.className.split(" ")[0];
497           if (actions[action] == 'disable') { // deactivate
498             this.disabled = true;
499             this.className = action + " disabled";
500           } else {
501             this.disabled = false;
502             this.className = action;
503             if(actions[action] === true) { // buttons
504               this.className = action + " on";
505             } else if(actions[action]){ // selects
506               this.value = actions[action];
507             }
508           }
509         });
510       }
511     }

514   /**
515    * Create a new Editor
516    * 
517    * An editor has a current mode and a textarea mode. Both are the same if you 
518    * edit the textarea directly (e.g. textile). In the wysiwyg mode you edit the
519    * html directly.
520    *
521    * @constructor
522    * @name Editor
523    * 
524    * @param {jQuery} textArea The textarea
525    * @param {Object} settings Editor specific settings
526    */
527   function Editor(textArea, settings) {
528     var editor = this, timer = 0, htmlDiv = settings.htmlDiv;
529 
530     this.setDataType(textArea.attr("class"));
531     this.settings = settings;
532 
533     if(!this.dataType) { return ;}
534 
535     function addKeyListeners(object, isTextarea){
536       object.keydown(function(e){
537         if(isTextarea || editor.is('wysiwyg')){
538           return editor.currentMode.pressed(editor, e.keyCode);
539         }
540       }).keyup(function(e){
541         if(isTextarea || editor.is('wysiwyg')){
542           return editor.currentMode.released(editor, e.keyCode);
543         }
544       }).mouseup(function(){
545         if(isTextarea || editor.is('wysiwyg')){
546           editor.focus();
547           return editor.currentMode.clicked(editor);
548         }
549       });
550     }
551 
552     // mouseup will catch all three mouse buttons. Since all keys move
553     // the cursor a check is necessary
554     this.textArea = textArea.bind("mouseup keyup", function() {
555       editor.checkState();
556       clearTimeout(timer);
557       timer = setTimeout(function(){
558         editor.currentMode.updatePreview(editor);
559       },1000);
560     });
561     addKeyListeners(textArea,true);
562 
563     if(htmlDiv){
564       htmlDiv.addClass('preview');
565     } else {
566       htmlDiv = $("<div class=\"preview\"></div>");
567     }
568     this.htmlDiv = htmlDiv.bind("mouseup keyup", function() {
569         if(editor.is("wysiwyg")) {
570           editor.checkState();
571         }
572       });
573     addKeyListeners(this.htmlDiv);
574     
575     this.toolbar = new Toolbar(this);
576     this.container = textArea.wrap("<div class=\"markupEditor\"></div>")
577       .parent().append(editor.htmlDiv).
578       prepend(this.toolbar.div);
579     textArea.wrap("<div class=\"textarea\">");
580 
581     activeEditors[numberOfEditors] = editor;
582     editor.id = numberOfEditors;
583     // console.log("init editor " + numberOfEditors);
584     numberOfEditors ++;
585   } // Editor

708   /**
709    * @namespace Holds all public methods
710    */
711   ME = {
712     /**
713      * Add a mode
714      *
715      * @param {String} modeId The id of the mode as referenced
716      * internally
717      * @param {Object} modeDefinition Defines methods and toolbaritems
718      * of the new Mode 
719      */
720     addMode: function(modeId, modeDefinition) {
721       var items = modeDefinition.items, constructor, supportedItems = globalItems.slice();
722       modeDefinition.id = modeId;
723       
724       if(items) {
725         for( item in items) {
726           if(items.hasOwnProperty(item) && item !== "default") {
727             supportedItems.push(item);
728             if(!toolbarItems[item]) {
729               constructor = items[item].options ? ToolbarSelect : ToolbarButton;
730               toolbarItems[item] = new constructor(item);
731             }
732             toolbarItems[item][modeId] = $.extend({name: item}, items["default"], items[item]);
733           }
734         }
735       }
736 
737       if(modeId !== 'wysiwyg'){
738         toolbarItems.changeDataMode.options.push([modeId, modeDefinition.name]);
739       }
740 
741       modeDefinition.supportedItems = supportedItems;
742       
743       return availableModes[modeId] = new Mode(modeDefinition);
744     },
745     /**
746      * Get the specified mode. Loads it if necessary
747      *
748      * @param {String} modeId The id of the mode (e.g. textile)
749      *
750      * @returns {Mode} The initialized mode
751      */
752     getMode: function(modeId) {
753       var mode = availableModes[modeId];
754       if (mode) {
755         return mode;
756       }

859   $.fn.markupEditor= function(method) {
860     var args = Array.prototype.slice.call(arguments, 1);
