172   ME.Toolbar.getSupportedItems = function(modeId, modeName, items){
173     var constructor,
174     supportedItems = globalItems.slice(); // global items (like
175     // save) are supported
176     
177     if(items) {
178       for( item in items) {
179         if(items.hasOwnProperty(item) && item !== "default") {
180           supportedItems.push(item);
181           if(!toolbarItems[item]) {
182             constructor = items[item].options ? ME.ToolbarSelect : ME.ToolbarButton;
183             toolbarItems[item] = new constructor(item);
184           }
185           toolbarItems[item][modeId] = $.extend({name: item}, items["default"], items[item]);
186         }
187       }
188     }
189 
190     if(modeId !== 'wysiwyg'){
191       toolbarItems.changeDataMode.options.push([modeId, modeName]);
192     }
193 
194     return supportedItems;
195   };

199      * Load the toolbar for the current mode. If a toolbar item is not
200      * supported, it will be hidden.
201      */
202     loadModeToolbar: function(editor){
203       var supportedItems = editor.currentMode.supportedItems,
204       oldVisibleItems = this.visibleItems,
205       newVisibleItems = [];
206       
207       // Optimize: better scheme. Calculate the differences between
208       // the modes once and use them here
209       this.div.children().each(function(){
210         var item = this.className.split(' ')[0], isAvailable = toolbarItems[item].isAvailable;
211         if(supportedItems.indexOf(item) != -1 && (!isAvailable || isAvailable(editor))){
212           if(!oldVisibleItems || oldVisibleItems.indexOf(item) == -1){
213             $(this).show();
214           }
215           newVisibleItems.push(item);
216         } else {
217           if(!oldVisibleItems || oldVisibleItems.indexOf(item) != -1){
218             $(this).hide();
219           }
220         }
221       });
222       this.visibleItems = newVisibleItems;
223     },
