 22   function getParagraphs() {
 23     var anchor, focus, paragraphs, matchIndex = -1;
 24 
 25     anchor = lastParentBeforePreview(startNode());
 26     focus = lastParentBeforePreview(endNode())[0];
 27     
 28     if(anchor[0] !== focus){
 29       paragraphs = anchor.nextAll().filter(function(i){
 30         if(this == focus){
 31           matchIndex = i;
 32         }
 33         if(matchIndex === -1 || matchIndex === i){
 34           return true;
 35         }
 36       }).add(anchor);
 37     } else {
 38       paragraphs = anchor;
 39     }
 40 
 41     return paragraphs;
 42   }

 50   function selectNodes(nodes,collapse){
 51     var firstNode = nodes[0], lastNode;
 52     if(nodes.length > 1){
 53       lastNode = nodes[nodes.length-1];
 54       range.setStart(firstNode,0);
 55       range.setEnd(lastNode,lastNode.childNodes.length);
 56     } else {
 57       range.selectNodeContents(firstNode);
 58     }
 59 
 60     if(collapse !== undefined){
 61       range.collapse(collapse);
 62     }
 63     selection.removeAllRanges();
 64     selection.addRange(range);
 65   }

 71     var contents, lines, $p, $list;
 72

104   /**
105    * Create a new Border with the following properties
106    *
107    * @property {String} nextProperty The property wich holds the next
108    * sibling
109    * @property {jQuery} ancestors
110    * @property {HTMLElement} block
111    * @property {HTMLElement} safeBlock
112    * @property {HTMLElement} borderNode
113    * Is null when border node is not found. That means the node has no
114    * more siblings
115    * @property {HTMLElement} node
116    * Equals borderNode if it exists, otherwise its the last sibling
117    *
118    * @param {HTMLElement} node
119    * @param {String} borderType The nodeName of the border node
120    * @param {String} nextProperty The property wich holds the next sibling
121    * @constructor
122    */
123   function Border(node, borderType, nextProperty){
124     this.nextProperty = nextProperty;
125     this.ancestors = node.parentsUntil(".preview");
126     this.block = this.ancestors[this.ancestors.length - 1] || node[0];
127     this.borderNode = this.ancestors[this.ancestors.length -2] || node[0];
128     while(this.borderNode){
129       this.node = this.borderNode;
130       if(!borderType || this.borderNode.nodeName.toLowerCase() === borderType){
131         break;
132       }
133       this.borderNode = this.borderNode[nextProperty];
134     }
135     this.safeBlock = this.borderNode ? this.block : this.block[nextProperty];
136   }

138   /**
139    * Create list items from node and it siblings and join them to the
140    * given list
141    *
142    * @param {HTMLElement} list
143    * @param {HTMLElement} node
144    */
145   function createList(list, node){
146     var li = document.createElement("li"), next;
147     
148     function pushItem(){
149       if(!/^\s*$/.test(li.textContent)){
150         list.append(li);
151       }
152       li = document.createElement('li');
153     }
154 
155     while(node !== null){
156       next = node.nextSibling;
157       if(/br/i.test(node.nodeName)){
158         pushItem();
159       } else if(/(p|h\d)/i.test(node.nodeName)) {
160         pushItem();
161         createList(list, node.firstChild);
162       } else if(/(o|u)l/i.test(node.nodeName)) {
163         pushItem();
164         $(node).children().appendTo(list);
165       } else if(/li/i.test(node.nodeName)) {
166         pushItem();
167         list.append(node);
168       } else {
169         li.appendChild(node);
170       }
171       node = next;
172     }
173     pushItem();
174   }

176   /**
177    * Add the items of an adjcent list node defined by border to the
178    * list.
179    *
180    * @param {Border} border
181    * @param {HTMLElement} list
182    */
183   function joinAdjacentList(border, list){
184     var children;
185     if(border.safeBlock && /ul/i.test(border.safeBlock.nodeName)){
186       next = border.safeBlock[border.nextProperty];
187 
188       children = $(border.safeBlock).remove().children();
189       if(border.nextProperty === 'previousSibling'){
190         children.prependTo(list);
191       } else {
192         children.appendTo(list);
193       }
194       border.safeBlock = next;
195     }
196   }

216   /**
217    * Check the position of the caret and adjust the container the
218    * caret is in when necessary
219    *
220    * @param {Integer} adjustment Adjust the current offset (needed for
221    * arrow keys
222    */
223   function checkCaret(adjustment){
224     var range = selection.getRangeAt(0),node, text;
225     function checkSibling(property, collapse){
226       while(!node[property]){
227         node = node.parentNode;
228       }
229       node = node[property];
230       if(node && !/br|h\d|p/i.test(node.nodeName)){
231         selectNodes([node], collapse);
232         return false;
233       }
234     }
235     if(range.collapsed){
236       node = range.startContainer;
237       if(node.nodeType == 3){ // Its a textnode
238         text = node.nodeValue;
239         if(range.startOffset + adjustment === 0 && /^ /.test(text)){
240           return checkSibling('previousSibling', false);
241         } else if(range.startOffset + adjustment === node.length && / $/.test(text)){
242           return checkSibling('nextSibling', true);
243         }
244       }
245     }
246   }

249    * @browserbug Firefox and Chrome
250    */
251   function pressedEnter(htmlDiv){
252     if(checkIfDeletedAll(htmlDiv,13) === false){
253       return false;
254     }
255     var lastSibling = true, node, range, endContainer,
256     block = editor.currentNodes.block;
257 
258     if(/h[1-5]/i.test(block.nodeName)){ // only headings
259       range = selection.getRangeAt(0);
260       node = endContainer = range.endContainer;
261 
262       // only the last sibling
263       while(node.parentNode !== htmlDiv[0]){
264         if(node.nextSibling){
265           lastSibling = false;
266           break;
267         }
268         node = node.parentNode;
269       }
270       
271       if(lastSibling
272          && range.endOffset === endContainer.textContent.length // at
273          // the end
274         ){
275         node = $("<p>").insertAfter(block);
276         selectNodes(node);
277         return false;
278       }
279     }

282   /**
283    * Handle the backspace key within lists and fixes a bug in Chrome.
284    *
285    * @browserbug Chrome
286    * Chrome tries to keep the styles whilst backspacing from say a
287    * paragraph to a heading. The content of the paragraph looks like a
288    * paragraph, but is a heading
289    *
290    * @param {Editor} editor The editor to work on
291    */
292   function pressedBackspace(editor, htmlDiv){
293     if(checkIfDeletedAll(htmlDiv,8) === false){
294       return false;
295     }
296     var children, atBeginningOfLI,
297     inFirstSibling = true,
298     block = editor.currentNodes.block,
299     list = editor.currentNodes.list,
300     prev = (list || block).previousSibling,
301     range = selection.getRangeAt(0),
302     node = range.startContainer;
303 
304     if(!range.collapsed || !prev || range.startOffset !== 0){
305       return true;
306     }
307 
308     // only the first sibling
309     while(node.parentNode !== htmlDiv[0]){
310       if(node.previousSibling){
311         inFirstSibling = false;
312         if(/li/i.test(node.nodeName)){
313           atBeginningOfLI = true;
314         } else {
315           return true;
316         }
317         break;
318       }
319       node = node.parentNode;
320     }
321 
322     if(inFirstSibling){
323       if(list){
324         node = node.firstChild;
325       }
326       node = $(node);
327       children = node.contents();
328 
329       $(prev).append(children);
330       node.remove();
331       selectNodes([children[0]], true);
332     } else if(list && atBeginningOfLI){
333       disableList(editor);
334     }
335     
336     return false;
337   }

345   function pressedDelete(editor, htmlDiv){
346     if(checkIfDeletedAll(htmlDiv,46) === false){
347       return false;
348     }
349     if(!$.browser.webkit){
350       return true;
351     }
352     var children,
353     block = editor.currentNodes.block,
354     list = editor.currentNodes.list,
355     next = (list || block).nextSibling,
356     range = selection.getRangeAt(0),
357     node = range.startContainer;
358 
359     if(!range.collapsed || !next || range.startOffset !== node.length){
360       return true;
361     }
362 
363     // only the first sibling
364     while(node.parentNode !== htmlDiv[0]){
365       if(node.nextSibling){
366         return true;
367       }
368       node = node.parentNode;
369     }
370 
371     if(list){
372       // append to last list item
373       node = node.lastChild;
374     }
375     if(/(u|o)l/i.test(next.nodeName)){
376       // swallow only the first list item
377       next = next.firstChild;
378     }
379     next = $(next);
380     children = next.contents();
381 
382     $(node).append(children);
383     next.remove();
384 
385     return false;
386   }

390    */
391   function checkIfDeletedAll(htmlDiv, keyCode, holdNeutralKey){
392     var range = selection.getRangeAt(0);
393     if(!$.browser.mozilla || holdNeutralKey || range.collapsed || ME.util.isNeutralKey(keyCode)){
394       return true;
395     }
396     var node, content;
397     content = range.extractContents();
398     // The second check is necessary, because extractContents might
399     // leave some empty tags when you manually select the whole div
400     // (the selection is always inside the block tags)
401     if(htmlDiv.is(":empty") || /^ *$/.test(htmlDiv.text())){
402       node = document.createElement(content.childNodes[0].nodeName);
403       htmlDiv.html(node);
404       selectNodes([node]);
405       if(!/^8|13|46$/.test("" + keyCode)){
406         // Pass non special keystrokes upwards
407         return true;
408       } else {
409         return false;
410       }
411     }
412   }
