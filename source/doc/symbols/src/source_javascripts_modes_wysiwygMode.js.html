 20   function getParagraphs() {
 21     var anchor, focus, paragraphs, matchIndex = -1;
 22 
 23     anchor = lastParentBeforePreview(startNode());
 24     focus = lastParentBeforePreview(endNode())[0];
 25     
 26     if(anchor[0] !== focus){
 27       paragraphs = anchor.nextAll().filter(function(i){
 28         if(this == focus){
 29           matchIndex = i;
 30         }
 31         if(matchIndex === -1 || matchIndex === i){
 32           return true;
 33         }
 34       }).add(anchor);
 35     } else {
 36       paragraphs = anchor;
 37     }
 38 
 39     return paragraphs;
 40   }

 48   function selectNodes(nodes,collapse){
 49     var firstNode = nodes[0], lastNode;
 50     if(nodes.length > 1){
 51       lastNode = nodes[nodes.length-1];
 52       range.setStart(firstNode,0);
 53       range.setEnd(lastNode,lastNode.childNodes.length);
 54     } else {
 55       range.selectNodeContents(firstNode);
 56     }
 57 
 58     if(collapse !== undefined){
 59       range.collapse(collapse);
 60     }
 61     selection.removeAllRanges();
 62     selection.addRange(range);
 63   }

 65   function handleList(editor, mode, target, listType){
 66     var contents, lines, $p, $list;
 67     
 68     if(/ on$/.test(target.className)){
 69       disableList(editor, mode);
 70     } else {
 71       contents = mode.getSelection('br');
 72       
 73       $list = $("<" + listType + ">");
 74       createList($list, contents.firstChild);
 75 
 76       joinAdjacentList(mode.leftBorder, $list);
 77       joinAdjacentList(mode.rightBorder, $list);
 78 
 79       mode.replaceSelection(editor, $list);
 80     }

 96   /**
 97    * Create a new Border with the following properties
 98    *
 99    * @property {String} nextProperty The property wich holds the next
100    * sibling
101    * @property {jQuery} ancestors
102    * @property {HTMLElement} block
103    * @property {HTMLElement} safeBlock
104    * @property {HTMLElement} borderNode
105    * Is null when border node is not found. That means the node has no
106    * more siblings
107    * @property {HTMLElement} node
108    * Equals borderNode if it exists, otherwise its the last sibling
109    *
110    * @param {HTMLElement} node
111    * @param {String} borderType The nodeName of the border node
112    * @param {String} nextProperty The property wich holds the next sibling
113    * @constructor
114    */
115   function Border(node, borderType, nextProperty){
116     this.nextProperty = nextProperty;
117     this.ancestors = node.parentsUntil(".preview");
118     this.block = this.ancestors[this.ancestors.length - 1] || node[0];
119     this.borderNode = this.ancestors[this.ancestors.length -2] || node[0];
120     while(this.borderNode){
121       this.node = this.borderNode;
122       if(!borderType || this.borderNode.nodeName.toLowerCase() === borderType){
123         break;
124       }
125       this.borderNode = this.borderNode[nextProperty];
126     }
127     this.safeBlock = this.borderNode ? this.block : this.block[nextProperty];
128   }

130   /**
131    * Create list items from node and it siblings and join them to the
132    * given list
133    *
134    * @param {HTMLElement} list
135    * @param {HTMLElement} node
136    */
137   function createList(list, node){
138     var li = document.createElement("li"), next;
139     
140     function pushItem(){
141       if(!/^\s*$/.test(li.textContent)){
142         list.append(li);
143       }
144       li = document.createElement('li');
145     }
146 
147     while(node !== null){
148       next = node.nextSibling;
149       if(/br/i.test(node.nodeName)){
150         pushItem();
151       } else if(/(p|h\d)/i.test(node.nodeName)) {
152         pushItem();
153         createList(list, node.firstChild);
154       } else if(/(o|u)l/i.test(node.nodeName)) {
155         pushItem();
156         $(node).children().appendTo(list);
157       } else if(/li/i.test(node.nodeName)) {
158         pushItem();
159         list.append(node);
160       } else {
161         li.appendChild(node);
162       }
163       node = next;
164     }
165     pushItem();
166   }

168   /**
169    * Add the items of an adjcent list node defined by border to the
170    * list.
171    *
172    * @param {Border} border
173    * @param {HTMLElement} list
174    */
175   function joinAdjacentList(border, list){
176     var children;
177     if(border.safeBlock && /ul/i.test(border.safeBlock.nodeName)){
178       next = border.safeBlock[border.nextProperty];
179 
180       children = $(border.safeBlock).remove().children();
181       if(border.nextProperty === 'previousSibling'){
182         children.prependTo(list);
183       } else {
184         children.appendTo(list);
185       }
186       border.safeBlock = next;
187     }
188   }

194    * @param {HTMLElement} node The list node

196   function addListItems(lines,node){
197     while(node){
198       if(/(o|u)l/i.test(node.nodeName)) {
199         addListItems(lines, node.firstChild);
200       } else if(/li/i.test(node.nodeName)) {
201         lines.push(node.innerHTML);
202       }
203       node = node.nextSibling;
204     }

208   /**
209    * Check the position of the caret and adjust the container the
210    * caret is in when necessary
211    *
212    * @param {Integer} adjustment Adjust the current offset (needed for
213    * arrow keys
214    */
215   function checkCaret(adjustment){
216     var range = selection.getRangeAt(0),node, text;
217     function checkSibling(property, collapse){
218       while(!node[property]){
219         node = node.parentNode;
220       }
221       node = node[property];
222       if(node && !/br|h\d|p/i.test(node.nodeName)){
223         selectNodes([node], collapse);
224         return false;
225       }
226     }
227     if(range.collapsed){
228       node = range.startContainer;
229       if(node.nodeType == 3){ // Its a textnode
230         text = node.nodeValue;
231         if(range.startOffset + adjustment === 0 && /^ /.test(text)){
232           return checkSibling('previousSibling', false);
233         } else if(range.startOffset + adjustment === node.length && / $/.test(text)){
234           return checkSibling('nextSibling', true);
235         }
236       }
237     }
238   }

241    * @browserbug Firefox and Chrome
242    */
243   function pressedEnter(htmlDiv){
244     if(checkIfDeletedAll(htmlDiv,13) === false){
245       return false;
246     }
247     var lastSibling = true, node, range, endContainer,
248     block = currentNodes.block;
249 
250     if(/h[1-5]/i.test(block.nodeName)){ // only headings
251       range = selection.getRangeAt(0);
252       node = endContainer = range.endContainer;
253 
254       // only the last sibling
255       while(node.parentNode !== htmlDiv[0]){
256         if(node.nextSibling){
257           lastSibling = false;
258           break;
259         }
260         node = node.parentNode;
261       }
262       
263       if(lastSibling
264          && range.endOffset === endContainer.textContent.length // at
265          // the end
266         ){
267         node = $("<p>").insertAfter(block);
268         selectNodes(node);
269         return false;
270       }
271     }

274   /**
275    * Handle the backspace key within lists and fixes a bug in Chrome.
276    *
277    * @browserbug Chrome
278    * Chrome tries to keep the styles whilst backspacing from say a
279    * paragraph to a heading. The content of the paragraph looks like a
280    * paragraph, but is a heading
281    */
282   function pressedBackspace(mode, htmlDiv, editor){
283     if(checkIfDeletedAll(htmlDiv,8) === false){
284       return false;
285     }
286     var children, atBeginningOfLI,
287     inFirstSibling = true,
288     block = currentNodes.block,
289     list = currentNodes.list,
290     prev = (list || block).previousSibling,
291     range = selection.getRangeAt(0),
292     node = range.startContainer;
293 
294     if(!range.collapsed || !prev || range.startOffset !== 0){
295       return true;
296     }
297 
298     // only the first sibling
299     while(node.parentNode !== htmlDiv[0]){
300       if(node.previousSibling){
301         inFirstSibling = false;
302         if(/li/i.test(node.nodeName)){
303           atBeginningOfLI = true;
304         } else {
305           return true;
306         }
307         break;
308       }
309       node = node.parentNode;
310     }
311 
312     if(inFirstSibling){
313       if(list){
314         node = node.firstChild;
315       }
316       node = $(node);
317       children = node.contents();
318 
319       $(prev).append(children);
320       node.remove();
321       selectNodes([children[0]], true);
322     } else if(list && atBeginningOfLI){
323       disableList(editor,mode);
324     }
325     
326     return false;
327   }

333   function pressedDelete(mode, htmlDiv, editor){
334     if(checkIfDeletedAll(htmlDiv,46) === false){
335       return false;
336     }
337     if(!$.browser.webkit){
338       return true;
339     }
340     var children,
341     block = currentNodes.block,
342     list = currentNodes.list,
343     next = (list || block).nextSibling,
344     range = selection.getRangeAt(0),
345     node = range.startContainer;
346 
347     if(!range.collapsed || !next || range.startOffset !== node.length){
348       return true;
349     }
350 
351     // only the first sibling
352     while(node.parentNode !== htmlDiv[0]){
353       if(node.nextSibling){
354         return true;
355       }
356       node = node.parentNode;
357     }
358 
359     if(list){
360       // append to last list item
361       node = node.lastChild;
362     }
363     if(/(u|o)l/i.test(next.nodeName)){
364       // swallow only the first list item
365       next = next.firstChild;
366     }
367     next = $(next);
368     children = next.contents();
369 
370     $(node).append(children);
371     next.remove();
372 
373     return false;
374   }

379   function checkIfDeletedAll(htmlDiv, keyCode, holdNeutralKey){
380     var range = selection.getRangeAt(0);
381     if(!$.browser.mozilla || holdNeutralKey || range.collapsed || ME.util.isNeutralKey(keyCode)){
382       return true;
383     }
384     var node, content;
385     content = range.extractContents();
386     // The second check is necessary, because extractContents might
387     // leave some empty tags when you manually select the whole div
388     // (the selection is always inside the block tags)
389     if(htmlDiv.is(":empty") || /^ *$/.test(htmlDiv.text())){
390       node = document.createElement(content.childNodes[0].nodeName);
391       htmlDiv.html(node);
392       selectNodes([node]);
393       if(!/^8|13|46$/.test("" + keyCode)){
394         // Pass non special keystrokes upwards
395         return true;
396       } else {
397         return false;
398       }
399     }
