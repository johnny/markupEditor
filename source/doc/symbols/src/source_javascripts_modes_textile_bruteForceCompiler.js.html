 12    * Builds the html tree and traces the common nodes inside a
 13    * selection

 16    * @namespace

 18   var builder = (function (){
 19     var stack, tracingStack, stackPosition, traceJustStarted, traceJustEnded, popping, pointer, sP, eP, tracing, lastTrace, unsuccessfulPush = false, stringLength,
 20     ignoredTags = ['li'],
 21     definableAttributes = {
 22       img: ['title','src'],
 23       a: ['href']
 24     };

 26     /**
 27      * Iterate over the attributes of the given tag and call the callback.
 28      * Used for comparison of the attributes of two nodes
 29      * 
 30      * @param {String} tag
 31      * @param {Function} callback
 32      */
 33     function iterateOverAttributes(tag, callback){
 34       var attributes = ['class'],i;
 35       if(definableAttributes[tag]){
 36         attributes = attributes.concat(definableAttributes[tag]);
 37       }
 38       for(i = attributes.length;i--;){
 39         callback(attributes[i]);
 40       }
 41     }

 43     /**
 44      * Test if the tracing stack can be moved down again with the given node.
 45      * 
 46      * @param {TraceNode} targetNode The node to test
 47      * 
 48      * @returns {Boolean}
 49      */
 50     function canMoveStackDown(targetNode){
 51       var equalTag = true, key, equalAttributes = true, blockTag = (stackPosition == -1), stackNode, i, l;
 52       if(blockTag){
 53         // List nodes should be weaker than a paragraph node
 54         if(/(o|u)l/.test(tracingStack[0].tag)){
 55           if(/(o|u)l/.test(targetNode.tag) && targetNode.tag != tracingStack[0].tag){
 56             targetNode = {tag: 'p'};
 57           }
 58           stackNode = tracingStack[0] = targetNode;
 59         } else {
 60           stackNode = tracingStack[0];
 61         }
 62       } else {
 63         for(i = stackPosition + 1, l = tracingStack.length; i < l; i++ ){
 64           if(tracingStack[i].tag === targetNode.tag){
 65             stackNode = tracingStack[i];
 66             tracingStack[i] = tracingStack[stackPosition + 1];
 67             tracingStack[stackPosition + 1] = stackNode;
 68             break;
 69           }
 70         }
 71       }
 72 
 73       if(stackNode){
 74         // console.log("checking attributes");
 75         if(stackNode.attributes){
 76           iterateOverAttributes(stackNode.tag,function(key){
 77             // console.log("checking",key);
 78             if(stackNode.attributes[key] !== targetNode.attributes[key]){
 79               // console.log("difference in ", key, stackNode.attributes[key], targetNode.attributes[key]);
 80               equalAttributes = false;
 81               delete stackNode.attributes[key];
 82             }
 83           });
 84         }
 85         
 86       }
 87       // console.log("!@!@!@", stackNode, targetNode, equalAttributes);
 88       return stackNode && (blockTag || equalAttributes);
 89     }

 91     /**
 92      * Create a new TraceNode form a given node
 93      * @constructor
 94      * 
 95      * @param {object} node 
 96      */
 97     function TraceNode(node){
 98       this.tag = node.tag;
 99       this.attributes = node.attributes;
100     }

102     /**
103      * Create a open-tag for the given node
104      * 
105      * @param {object} node
106      * 
107      * @return {String}
108      */
109     function htmlOpenTag(node){
110       var attributeString = "";
111       for(attr in node.attributes){
112         if(node.attributes.hasOwnProperty(attr)){
113           attributeString += " " + attr + "=\"" + node.attributes[attr] + "\"";
114         }
115       }
116       return "<" + node.tag + attributeString + ">";
117     }

119     /**
120      * Warning! this supposes, that there is only one instance of any tag in
121      * the stack
122      * 
123      * @param {String} tag
124      * 
125      * @returns {Integer} The stack position of the given tag
126      */
127     function getStackPositionOf(tag){
128       var i;
129       for(i = stack.length;i--; ){
130         if(stack[i].tag === tag){
131           return i;
132         }
133       }
134     }

136     /**
137      * Start the tracing
138      */
139     function startTrace(){
140       var length = stack.length, i, numberOfIgnoredTags = 0;
141       // console.log("################################## startTrace");
142       tracing = true;
143       traceJustStarted = true;
144       for(i=1;i<length;i++){
145         if(ignoredTags.indexOf(stack[i].tag) == -1){
146           tracingStack[i-1 - numberOfIgnoredTags] = new TraceNode(stack[i]);
147         } else {
148           numberOfIgnoredTags += 1;
149         }
150       }
151       // console.log(tracingStack.length);
152       stackPosition = tracingStack.length -1;
153     }
154     
155     /**
156      * End the tracing
157      */
158     function endTrace(){
159       // console.log("#################################### endTrace");
160       // console.log(tracingStack.length);
161       lastTrace = false;
162       tracing = false;
163     }

166      * @lends Builder
167      */
168     return {
169       /**
170        * Initialize Builder for normal operation
171        */
172       init: function(){
173         stack = [{content:""}];
174       },
175       /**
176        * Initialize Builder for tracing operation
177        */
178       initTrace: function(startPosition, endPosition, lengthOfString){
179         tracingStack = [];
180         tracing = undefined;
181         pointer = 0;
182         sP = startPosition;
183         eP = endPosition;
184         stringLength = lengthOfString;
185       },
186       /**
187        * Definitly ends the trace
188        */
189       finalizeTrace: function(){
190         if(tracing){
191           endTrace();
192         }
193       },
194       /**
195        * Advance the pointer by the given amount.
196        * Handles starting and ending of the trace
197        * 
198        * @param {Integer} advanceAmount
199        * @param {Boolean} forceEndTrace wether to forcably end the trace
200        */
201       advancePointer: function(advanceAmount, forceEndTrace){
202         pointer += advanceAmount;
203         // console.log("pointer",pointer,"startPointer", sP, "endPointer", eP);
204         if(tracing === undefined && (pointer > sP || pointer == stringLength)){
205           startTrace();
206         }
207         if(tracing && pointer > eP){
208           if(lastTrace || forceEndTrace){
209             endTrace();
210           } else {
211             // console.log("set last trace");
212             lastTrace = true;
213           }
214         }
215       },
216       /**
217        * Push a tag to the build stack
218        * 
219        * @param {String} tag
220        * @param {Object} attributes
221        */
222       pushTag: function(tag, attributes){
223         var node = {tag: tag,
224                     attributes: attributes || {},
225                     content: ""};
226         // console.log("open tag", node);
227         stack.push(node);
228         if(tracing && ignoredTags.indexOf(tag) == -1){
229           if(traceJustStarted){
230             // console.log("inserting node ", node);
231             tracingStack[stackPosition+1] = new TraceNode(node);
232             stackPosition += 1;
233           }
234           else if(tracingStack[stackPosition+1]){
235             if(canMoveStackDown(node)){
236               stackPosition += 1;
237             } else {
238               unsuccessfulPush = true;
239             }
240           }
241           // console.log("stackPosition " + stackPosition);
242         }
243       },
244       /**
245        * Close a tag. If a tag is given find it and close it. Otherwise the top
246        * tag is closed
247        * 
248        * @param {String} [tag]
249        */
250       closeTag: function(tag){
251         var removedNode, i;
252         if(tag){
253           i = getStackPositionOf(tag);
254           removedNode = stack.splice(i,1)[0];
255         } else {
256           removedNode = stack.pop();
257         }
258 
259         // console.log("closing", removedNode);
260         if(tracing && ignoredTags.indexOf(tag) == -1){
261           traceJustStarted = false;
262           if(unsuccessfulPush){
263             // console.log("slicing because of difference "+stackPosition);
264             tracingStack = tracingStack.slice(0,stackPosition+1);
265             unsuccessfulPush = false;
266           }
267           // console.log(tracingStack, stackPosition, removedNode);
268           if(tracingStack[stackPosition].tag === removedNode.tag){
269             // console.log("moving down");
270             stackPosition -= 1;
271             // console.log("stackPosition " + stackPosition);
272           }
273           popping = true;
274         }
275 
276         // The part of the content, which belongs exclusively to the
277         // current tag must be commited
278         this.pushString(htmlOpenTag(removedNode) + removedNode.content, stack[i-1]);
279         // The tag needs to be closed at the top, where the current string insertion occurs
280         this.pushString("</"+removedNode.tag+">"); 
281         popping = false;
282       },
283       /**
284        * Closes all textile markup that ends at a line break.
285        * For example "*" or "_"
286        */
287       popLineEnd: function(){
288         var surpressLineBreak = false, partialMarkup = {b: "*", i: "_"}, node;
289         // If the need arises to search deeper think of the correct
290         // order in which to append the content
291         while("a,i,b,li".indexOf(stack[stack.length - 1].tag) != -1){
292           if("li" === stack[stack.length-1].tag){
293             this.closeTag();
294             surpressLineBreak = true;
295           } else {
296             node = stack.pop();
297             this.pushString(partialMarkup[node.tag]+node.content);
298           }
299         }
300         return surpressLineBreak;
301       },
302       /**
303        * Closes all open tags
304        */
305       popParagraphEnd: function(){
306         while(stack.length > 1){
307           this.closeTag();
308         }
309       },
310       /**
311        * Add a string to the given node or the top node.
312        * 
313        * @param {String} string
314        * @param {Object} [node] defaults to the top node in the stack
315        */
316       pushString: function(string, node){
317         if(!node){
318           node = stack[stack.length - 1];
319         }
320         node.content += string;
321         // console.log("pushing", string, "to", node);
322         if(!/^([ ]+|<br\/>)?$/.test(string)){ // Ignore whitespace
323           // during tracing
324           if(traceJustStarted){
325             traceJustStarted = false;
326           }
327           if(tracing && !popping && tracingStack[stackPosition+1]){
328             // console.log("cutting because of String " + stackPosition);
329             // console.log(tracingStack.length);
330             tracingStack = tracingStack.slice(0,stackPosition + 1);
331             // console.log(tracingStack.length);
332           }
333         }
334       },

342       isOpen: function(tag){

372   function advance(regexp){
373     var match = regexp.exec(text),whitespaceLength, matchLength;
374     if(match){
375       whitespaceLength = /^\s*/.exec(match[0])[0].length;
376       matchLength = match[0].length;
377       if(whitespaceLength){
378         builder.advancePointer(whitespaceLength, true);
379       }
380       if(matchLength - whitespaceLength){
381         builder.advancePointer(matchLength - whitespaceLength);
382       }
383       // console.log("@@@@ matched: ", match, regexp);
384       text = text.slice(matchLength);
385       return match || true;
386     }
387   }
388   function next(regexp){
389     return regexp.test(text);
390   }
391   
392   function parseParagraphs(){
393     var match;
394     while(!/^\s*$/.test(text)){
395       match = advance(/^\s*(h\d|p|bq)(\(([^#\)]*)(#[^\)]+)?\))?\. /);
396       if(match){
397         var attributes = {};
398         if(match[3]){
399           attributes["class"] = match[3];
400         }
401         if(match[4]){
402           attributes.id = match[4];
403         }
404 
405         builder.pushTag(match[1], attributes);
406       }
407       parseLines();
408       builder.popParagraphEnd();
409     }
410   }

412   function parseLines(){
413     while(!advance(/^\n/) && !next(/^\s*$/)){ // check for a double linebreak
414       parseLineStart();
415     }
416   }
417   
418   function parseLineStart(){
419     var match;
420     if(advance(/^ *\* /)){
421       if(!builder.isOpen("ul")){ // this won't work for nested uls,
422         // solve with lookahead
423         builder.closeBlockTag();
424         builder.pushTag("ul");
425       }
426       builder.pushTag("li");
427     } else if(advance(/^ *# /)){
428       if(!builder.isOpen("ol")){ // this won't work for nested uls
429         builder.closeBlockTag();
430         builder.pushTag("ol");
431       }
432       builder.pushTag("li");
433     } else {
434       while(builder.isOpen("ul") || builder.isOpen("ol")){
435         builder.closeTag();
436       }
437       if(!builder.blockTagIsOpen()){
438         builder.pushTag("p");
439       }
440     }
441     // Eat Whitespace at the beginning of the Line after the tag
442     // TODO remove this and eat the whitespace at the beginning of
443     // each regexp
444     match = advance(/^ */);
445     builder.pushString(match[0]);
446     parseLine();
447   }
448   function parseLine(){
449     var match, isInListOrTable;
450     while(true){
451       // Italic start
452       if(match = advance(/^_(?=[^ \n]+)/)){
453         if(builder.isOpen("i")){
454           builder.pushString("_");
455         } else {
456           builder.pushTag("i");
457         }
458       }
459       // bold start
460       else if(match = advance(/^\*(?=[^ \n]+)/)){
461         if(builder.isOpen("b")){
462           builder.pushString("*");
463         } else {
464           builder.pushTag("b");
465         }
466       }
467       // italic end
468       else if(match = advance(/^([^ \n]+)_( +|(?=\n|$))/)){
469         if(builder.isOpen("i")){
470           builder.pushString(match[1]);
471           builder.closeTag("i");
472           builder.pushString(match[2]);
473         } else {
474           builder.pushString(match[1] + "_" + match[2]);
475         }
476       }
477       // bold end
478       else if(match = advance(/^([^ \n]+)\*( +|(?=\n|$))/)){
479         if(builder.isOpen("b")){
480           builder.pushString(match[1]);
481           builder.closeTag("b");
482           builder.pushString(match[2]);
483         } else {
484           builder.pushString(match[1] + "*" + match[2]);
485         }
486       }
487       // link
488       else if(match = advance(/^( *)"([^"]*)":([^ \n]+)/)) {
489         builder.pushString(match[1]);
490         builder.pushTag("a", {href: match[3]});
491         builder.pushString(match[2]);
492         builder.closeTag();
493       }
494       // Image
495       else if(match = advance(/^( *)!([^!\(]+)(\(([^\)]*)\))?!(:([^ \n]+))?/)) {
496         builder.pushString(match[1]);
497         if(match[6]){
498           builder.pushTag("a", {href: match[6]});
499         }
500         var attributes = {src: match[2]};
501         if(match[4]){
502           attributes.title = match[4];
503         }
504         builder.pushTag("img", attributes);
505         builder.closeTag();
506         if(match[6]){
507           builder.closeTag();
508         }
509       }
510       // word or blanks
511       else if(match = advance(/^([^ \n]+)/)){
512         builder.pushString(match[1]);
513       }
514       // blanks
515       else if(match = advance(/^( +)/)){
516         builder.pushString(match[1]);
517       }
518       // end of line
519       else {
520         advance(/^\n/);
521         // If in List for example, surpress line break
522         isInListOrTable = builder.popLineEnd();
523         if(!isInListOrTable && !next(/^\s*(\n|$|[\*#] )/)){
524           builder.pushString("<br/>");
525         }
526         return;
527       }
528     }
529   }

537   textileCompiler = {
538     /**
539      * Compile the given text to html
540      *
541      * @param {String} textToCompile A textile string
542      *
543      * @returns {String} The html
544      */
545     compile: function(textToCompile){
546       builder.init();
547       text = textToCompile;
548       parseParagraphs();
549       return builder.toHtml();
550     },
551     /**
552      * Trace the common nodes between the given boundaries
553      *
554      * @param {String} textToCompile A textile string
555      * @param {Integer} startTrace Index of textToCompile where to
556      * start the trace
557      * @param {Integer} endTrace Index of textToCompile where to
558      * end the trace
559      *
560      * @returns {Object} A stack of the common nodes
561      */
562     trace: function(textToCompile, startTrace, endTrace){
563       builder.initTrace(startTrace, endTrace, textToCompile.length);
564       this.compile(textToCompile);
565       builder.finalizeTrace();
566       return builder.getTrace();
567     }
